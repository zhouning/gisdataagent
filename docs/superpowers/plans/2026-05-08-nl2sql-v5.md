> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Turn the v4 manuscript into an IJGIS-submittable v5 that clears both 2026-05-08 review reports' blockers — word count ≤7800 words (main text + refs), Abstract ≤220 words, zero v3/v4 leakage, zero "done but described as future work" contradictions, factually correct related-work descriptions, one open-source LLM cross-model ablation to answer the model-family-generalisation concern, and a DIN-SQL held-out + Robustness-paired external-baseline round to close the last first-revision candidate.

A final Phase 4 assembles the v5 submission bundle (new directory `submission/nl2semantic2sql_v5/`), re-compiles PDFs, verifies word count with `texcount`, writes the v5 response-to-reviewers, and commits in code / paper-separated commits per the paper-not-in-github policy.

All evals use the 125-question Chongqing PostGIS benchmark + BIRD mini_dev + the paired McNemar tooling already in `scripts/nl2sql_bench_cq/`.

```python
"""Build a 30-question stratified subset of the 85q GIS Spatial benchmark for
cross-model-family ablation. Stratify by (difficulty, category) pair, sampling
proportional counts that round to at least 1 per non-empty cell.

Output: benchmarks/gis_spatial_30q_subset.json (same record format as source).
"""
from __future__ import annotations
import json
import random
from collections import defaultdict
from pathlib import Path

SRC = Path("D:/adk/benchmarks/chongqing_geo_nl2sql_full_benchmark_v2.json")
OUT = Path("D:/adk/benchmarks/gis_spatial_30q_subset.json")
TARGET = 30
SEED = 20260508

def main() -> None:
    rows = json.loads(SRC.read_text(encoding="utf-8"))
    # Only keep the 85 Spatial questions; the benchmark also contains
    # robustness items; filter by id prefix if needed.
    spatial = [r for r in rows if r.get("id", "").startswith("CQ_GEO")]
    if len(spatial) != 85:
        print(f"[warn] expected 85 spatial, got {len(spatial)}")
    by_cell = defaultdict(list)
    for r in spatial:
        by_cell[(r.get("difficulty", "?"), r.get("category", "?"))].append(r)
    rng = random.Random(SEED)
    picked: list[dict] = []
    cells = sorted(by_cell.items(), key=lambda kv: -len(kv[1]))
    total = len(spatial)
    for (diff, cat), items in cells:
        share = max(1, round(len(items) / total * TARGET))
        share = min(share, len(items))
        rng.shuffle(items)
        picked.extend(items[:share])
    # Trim or pad to exactly TARGET
    if len(picked) > TARGET:
        picked = picked[:TARGET]
    elif len(picked) < TARGET:
        leftover = [r for cell in by_cell.values() for r in cell if r not in picked]
        rng.shuffle(leftover)
        picked.extend(leftover[:TARGET - len(picked)])
    picked.sort(key=lambda r: r.get("id", ""))
    OUT.write_text(json.dumps(picked, ensure_ascii=False, indent=2), encoding="utf-8")
    print(f"wrote {OUT} with {len(picked)} rows")
    from collections import Counter
    print("by difficulty:", Counter(r.get("difficulty") for r in picked))
    print("by category:",   Counter(r.get("category") for r in picked))

if __name__ == "__main__":
    main()
```

"""Run GIS Spatial 30q in baseline and full modes under DeepSeek, then under
Gemini (for a paired within-model comparison + a head-to-head comparison).

The point of this run is not to beat Gemini's Spatial EX but to demonstrate
that semantic grounding's direction and statistical signal generalise to a

  PYTHONPATH=/d/adk PYTHONIOENCODING=utf-8 \
    /d/adk/.venv/Scripts/python.exe scripts/nl2sql_bench_cq/run_open_source_ablation.py
"""
from __future__ import annotations

import asyncio
import json
import os
import sys
from datetime import datetime
from pathlib import Path

ROOT = Path(__file__).resolve().parents[2]
sys.path.insert(0, str(ROOT))
sys.path.insert(0, str(ROOT / "scripts" / "nl2sql_bench_cq"))

BENCH = ROOT / "benchmarks" / "gis_spatial_30q_subset.json"
OUT_ROOT = ROOT / "data_agent" / "nl2sql_eval_results"

def mcnemar_exact_two_sided(b: int, c: int) -> float:
    from math import comb
    n = b + c
    if n == 0:
        return 1.0
    k = min(b, c)
    p_one = sum(comb(n, i) for i in range(k + 1)) / (2 ** n)
    return min(1.0, 2 * p_one)

async def run_one_family(family: str, qs: list[dict], out_dir: Path) -> dict[str, list[dict]]:
    """family in {'gemini', 'deepseek'}; returns {'baseline':[...], 'full':[...]}"""
    # Swap env flags before importing the cq runner
    if family == "deepseek":
        os.environ["NL2SQL_FORCE_DEEPSEEK"] = "1"
        os.environ["NL2SQL_AGENT_MODEL"] = "deepseek-v4-flash"
    else:
        os.environ.pop("NL2SQL_FORCE_DEEPSEEK", None)
        os.environ["NL2SQL_AGENT_MODEL"] = "gemini-2.5-flash"

    # Use the chongqing-geo eval runner helpers
    from run_cq_eval import run_one, _init_runtime  # noqa: E402
    _init_runtime()

    out = {"baseline": [], "full": []}
    for mode in ("baseline", "full"):
        for i, q in enumerate(qs, 1):
            try:
                rec = await asyncio.wait_for(run_one(q, mode), timeout=90)
            except Exception as e:
                rec = {
                    "qid": q["id"], "ex": 0, "gen_status": "exception",
                    "gen_error": str(e), "family": family, "mode": mode,
                }
            rec["family"] = family
            rec["mode"] = mode
            out[mode].append(rec)
            print(f"  [{family}/{mode}] {i}/{len(qs)}: {q.get('id')} ex={rec.get('ex')}")
    return out

async def main() -> int:
    qs = json.loads(BENCH.read_text(encoding="utf-8"))
    ts = datetime.now().strftime("%Y-%m-%d_%H%M%S")
    out_dir = OUT_ROOT / f"open_source_ablation_{ts}"
    out_dir.mkdir(parents=True, exist_ok=True)

    deepseek = await run_one_family("deepseek", qs, out_dir)
    gemini   = await run_one_family("gemini",   qs, out_dir)

    def ex_by_qid(recs):
        return {r["qid"]: (1 if r.get("ex") else 0) for r in recs}

    def paired(a_ex, b_ex, label_a, label_b):
        qids = sorted(a_ex.keys() & b_ex.keys())
        tp=fp=fn=tn=0
        for q in qids:
            pa, pb = a_ex[q], b_ex[q]
            if pa and pb: tp+=1
            elif pa and not pb: fp+=1
            elif not pa and pb: fn+=1
            else: tn+=1
        p = mcnemar_exact_two_sided(fp, fn)
        return {
            "n": len(qids), "label_a": label_a, "label_b": label_b,
            "ex_a": round(sum(a_ex.values())/max(1,len(a_ex)), 4),
            "ex_b": round(sum(b_ex.values())/max(1,len(b_ex)), 4),
            "discordant_a1b0": fp, "discordant_a0b1": fn,
            "mcnemar_p_two_sided_exact": round(p, 4),
        }

    ds_base = ex_by_qid(deepseek["baseline"])
    ds_full = ex_by_qid(deepseek["full"])
    gm_base = ex_by_qid(gemini["baseline"])
    gm_full = ex_by_qid(gemini["full"])

    report = {
        "generated_at": datetime.now().isoformat(),
        "benchmark": str(BENCH.relative_to(ROOT)),
        "n_questions": len(qs),
        "pairs": [
            paired(ds_base, ds_full, "deepseek_baseline", "deepseek_full"),
            paired(gm_base, gm_full, "gemini_baseline",   "gemini_full"),
            paired(ds_full, gm_full, "deepseek_full",     "gemini_full"),
        ],
        "deepseek": deepseek,
        "gemini":   gemini,
    }
    out_path = out_dir / "open_source_ablation_report.json"
    out_path.write_text(json.dumps(report, indent=2, ensure_ascii=False, default=str),
                        encoding="utf-8")
    print(f"\nReport: {out_path}")
    for pr in report["pairs"]:
        print(f"  {pr['label_a']} EX={pr['ex_a']}  vs  {pr['label_b']} EX={pr['ex_b']}  "
              f"p={pr['mcnemar_p_two_sided_exact']}  discordant {pr['discordant_a1b0']}/{pr['discordant_a0b1']}")
    return 0

if __name__ == "__main__":
    raise SystemExit(asyncio.run(main()))
```

        is_correct = is_valid and gold_res["status"] == "ok" and \
                     compare_results(gold_res["rows"], pred_res["rows"])
        rec = {
            "qid": q.get("question_id") or q.get("id"),
            "db_id": db_id, "difficulty": q.get("difficulty", "?"),
            "question": q["question"], "gold_sql": gold_sql, "pred_sql": pred_sql,
            "ex": 1 if is_correct else 0, "valid": 1 if is_valid else 0,
        }
        recs.append(rec)
        print(f"  [{i}/{len(qs)}] ex={rec['ex']} valid={rec['valid']} qid={rec['qid']}")

    report = {
        "generated_at": datetime.now().isoformat(),
        "benchmark": args.bench,
        "n": len(recs),
        "ex": round(sum(r["ex"] for r in recs) / max(1, len(recs)), 4),
        "valid": round(sum(r["valid"] for r in recs) / max(1, len(recs)), 4),
        "records": recs,
    }
    (out_dir / "din_sql_fast_report.json").write_text(
        json.dumps(report, indent=2, ensure_ascii=False, default=str), encoding="utf-8"
    )
    print(f"\nDIN-SQL EX={report['ex']} Valid={report['valid']}")
    return 0

if __name__ == "__main__":
    raise SystemExit(asyncio.run(main()))
```

import json, pathlib
from math import comb

def p2s(b, c):
    n = b+c
    if n == 0: return 1.0

    return min(1.0, 2 * sum(comb(n,i) for i in range(k+1)) / (2**n))

qids = sorted(set(din_ex) & set(our_ex))
b=c=0
for q in qids:
