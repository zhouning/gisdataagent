```python
# data_agent/semantic_graph.py
"""Metadata graph G=(V,E) for PostGIS semantic grounding.

V = E_geo (geographic entities) ∪ D (dimensions) ∪ M (measures) ∪
    I (intent tags) ∪ P (OGC predicates) ∪ C (safety/unit constraints).

E = E_hasGeom ∪ E_fk ∪ E_topo ∪ E_metric ∪ E_knn ∪
    E_intent_routes ∪ E_safety ∪ E_unit.

This is the executable subset of OGC SFA (06-104r4) the agent grounds against.
Cite `semantic_graph` in paper §3.1 as the metadata-graph abstraction suggested
by the v2 peer-review report §3.A.
"""
from __future__ import annotations

from dataclasses import dataclass
from enum import Enum
from typing import Optional

import networkx as nx


class NodeKind(str, Enum):
    GEO_ENTITY = "geo_entity"
    DIMENSION = "dimension"
    MEASURE = "measure"
    INTENT = "intent"
    PREDICATE = "predicate"
    CONSTRAINT = "constraint"


class EdgeKind(str, Enum):
    HAS_GEOMETRY = "has_geometry"
    FK = "fk"
    TOPOLOGICAL = "topological"
    METRIC = "metric"
    KNN = "knn"
    INTENT_ROUTES = "intent_routes"
    SAFETY_CONSTRAINT = "safety_constraint"
    UNIT_RULE = "unit_rule"


@dataclass
class SemanticGraph:
    graph: nx.MultiDiGraph = None

    def __post_init__(self):
        if self.graph is None:
            self.graph = nx.MultiDiGraph()
```

```python
    def add_geo_entity(self, table, geom_col, srid, geom_type):
        self.graph.add_node(table, kind=NodeKind.GEO_ENTITY,
            geom_col=geom_col, srid=srid, geom_type=geom_type)

        self.graph.add_edge(predicate, key, kind=EdgeKind.UNIT_RULE,
            required_cast=required_cast, yields_unit=yields_unit,
            srid_range=srid_range)

        lines = ["graph LR"]
        for src, dst, data in sorted(self.graph.edges(data=True)):
            lines.append(f"    {src} --> {dst}")
```

```python
def build_semantic_graph(schema: str = "public",
                        table_prefix: Optional[str] = None) -> SemanticGraph:
    """Materialize G=(V,E) from live PostGIS catalog + FK + static OGC vocab.

    V: geo_entities from geometry_columns; intents and predicates from
       _register_ogc_vocab(); constraints synthesized from SRID/cast rules.
    E: has_geometry (table↔geom_col), fk (from information_schema.key_column_usage),
       topological/metric/knn (predicates↔entities), intent_routes (static),
       unit_rule (SRID range→cast).
    """
    from sqlalchemy import text
    from data_agent.db_engine import get_engine

    g = SemanticGraph()
    _register_ogc_vocab(g)

    engine = get_engine()
    with engine.connect() as conn:
        q = text("""
            SELECT f_table_name, f_geometry_column, srid, type
            FROM geometry_columns
            WHERE f_table_schema = :schema
            ORDER BY f_table_name
        """)
        rows = conn.execute(q, {"schema": schema}).fetchall()

    for table, geom_col, srid, geom_type in rows:
        if table_prefix and not table.startswith(table_prefix):
            continue
        if srid == 0:  # grid tables have no CRS
            continue
        g.add_geo_entity(table, geom_col, int(srid), geom_type)

    return g


def _register_ogc_vocab(g: SemanticGraph) -> None:
    """Static registry of OGC SFA predicates + intent routing + unit rules."""
    topo = ["ST_Intersects", "ST_Contains", "ST_Within",
            "ST_Touches", "ST_Crosses", "ST_Overlaps", "ST_Equals"]
    metric = ["ST_Distance", "ST_Length", "ST_Area", "ST_DWithin"]
    knn = ["<->"]
    for p in topo: g.add_predicate(p, "topological")
    for p in metric: g.add_predicate(p, "metric")
    for p in knn: g.add_predicate(p, "knn")
    for i in ["attribute_filter", "aggregation", "spatial_join",
              "spatial_measurement", "knn", "preview_listing"]:
        g.add_intent(i)
    for p in topo: g.add_intent_route("spatial_join", p)
    for p in metric: g.add_intent_route("spatial_measurement", p)
    g.add_intent_route("knn", "<->")
    for p in metric:
        g.add_unit_rule(p, (4326, 4326), "geography", "metre")
```

```python
    from data_agent.semantic_graph import build_semantic_graph, EdgeKind
```

```python
def explain_row_estimate(sql: str, timeout_ms: int = 2000) -> Optional[int]:
    """Ask the PostgreSQL planner for estimated row count via EXPLAIN (FORMAT JSON).

    Returns None on any error (parse, timeout, connection). Always safe to call
    from untrusted SQL: EXPLAIN itself does not execute the query.
    """
    try:
        from data_agent.db_engine import get_engine
        from sqlalchemy import text
        engine = get_engine()
        with engine.connect() as conn:
            conn.execute(text("SET LOCAL statement_timeout = :t"),
                         {"t": int(timeout_ms)})
            res = conn.execute(text(f"EXPLAIN (FORMAT JSON) {sql}")).fetchone()
        if not res or not res[0]:
            return None
        plan = res[0][0]["Plan"]
        return int(plan.get("Plan Rows", 0))
    except Exception:
        return None
```

```python
"""Emit a representativeness table for the 85q GIS Spatial benchmark.

Counts:
  - OGC SFA predicate family (topological/metric/KNN/aggregation)
  - Specific PostGIS function frequency (ST_Intersects, ST_DWithin, ...)
  - SRID/CRS distribution (inferred from referenced tables via geometry_columns)
  - Join multiplicity (single-table / 2-table / 3+-table)
  - Difficulty stratification

import json, re
from collections import Counter
from pathlib import Path

BENCH = Path("D:/adk/benchmarks/chongqing_geo_nl2sql_full_benchmark.json")
OUT_TEX = Path("D:/adk/submission/nl2semantic2sql_v4/table_benchmark_profile.tex")

POSTGIS_FNS = ["ST_Intersects","ST_Contains","ST_Within","ST_Touches",
               "ST_Crosses","ST_Overlaps","ST_Distance","ST_Length",
               "ST_Area","ST_DWithin","ST_Buffer","ST_Transform"]
KNN_OP = "<->"

def classify_predicate_family(sql: str) -> list[str]:
    up = sql.upper()
    fams = []
    if any(f in up for f in ["ST_INTERSECTS","ST_CONTAINS","ST_WITHIN","ST_TOUCHES","ST_CROSSES","ST_OVERLAPS","ST_EQUALS"]):
        fams.append("topological")
    if any(f in up for f in ["ST_DISTANCE","ST_LENGTH","ST_AREA","ST_DWITHIN"]):
        fams.append("metric")
    if "<->" in sql:
        fams.append("knn")
    if re.search(r"\bGROUP BY\b", up) or re.search(r"\b(SUM|COUNT|AVG|MIN|MAX)\s*\(", up):
        fams.append("aggregation")
    return fams or ["plain"]

def main():
    qs = json.loads(BENCH.read_text(encoding="utf-8"))
    spatial = [q for q in qs if q["difficulty"] != "Robustness"]
    fam_counter = Counter()
    fn_counter = Counter()
    diff_counter = Counter()
    for q in spatial:
        gold = q.get("golden_sql", "")
        for f in classify_predicate_family(gold): fam_counter[f] += 1
        for fn in POSTGIS_FNS:
            if fn.upper() in gold.upper(): fn_counter[fn] += 1
        if KNN_OP in gold: fn_counter[KNN_OP] += 1
        diff_counter[q["difficulty"]] += 1

    lines = [r"\begin{tabular}{lr}", r"\toprule", r"Category & Count \\", r"\midrule",

    lines.append(r"\multicolumn{2}{l}{\emph{PostGIS function frequency}} \\")
    for fn, c in fn_counter.most_common():
        lines.append(f"\\quad \\texttt{{{fn}}} & {c} \\\\")
    lines.append(r"\multicolumn{2}{l}{\emph{Difficulty}} \\")
    for d, c in sorted(diff_counter.items()):
        lines.append(f"\\quad {d} & {c} \\\\")
    lines.append(f"\\midrule Total Spatial questions & {len(spatial)} \\\\")
    lines.append(r"\bottomrule"); lines.append(r"\end{tabular}")
    OUT_TEX.write_text("\n".join(lines), encoding="utf-8")
```

```latex
\subsection{Formal scope: semantic layer as metadata graph \texorpdfstring{$G=(V,E)$}{G=(V,E)}}
\label{sec:formal-scope}

Following the metadata-graph abstraction suggested in prior reviewer feedback
for NL-to-spatial-database systems, we formalise the semantic layer as a
directed multigraph $G = (V, E)$ that encodes the executable subset of OGC
Simple Feature Access~\cite{ogcsfs} required for PostGIS grounding.

\noindent\textbf{Vertices.} $V = E_{geo} \cup D \cup M \cup I \cup P \cup C$:
\begin{itemize}
\item $E_{geo}$ — geographic entities; each vertex is a PostGIS table tagged
  with (geometry column, SRID, geometry type).
\item $D, M$ — dimension and measure vertices for the warehouse track
  (BIRD-style schemas).
\item $I$ — intent tags
  $\{$attribute\_filter, aggregation, spatial\_join, spatial\_measurement,
  knn, preview\_listing$\}$.
\item $P$ — OGC SFA predicate vertices, partitioned into
  topological $\{\mathrm{ST\_Intersects}, \mathrm{ST\_Contains}, \mathrm{ST\_Within},
  \mathrm{ST\_Touches}, \mathrm{ST\_Crosses}, \mathrm{ST\_Overlaps}, \mathrm{ST\_Equals}\}$,
  metric $\{\mathrm{ST\_Distance}, \mathrm{ST\_Length}, \mathrm{ST\_Area}, \mathrm{ST\_DWithin}\}$,
  and KNN $\{$\texttt{<->}$\}$.
\item $C$ — constraint vertices (SRID ranges, unit rules, safety/refusal).
\end{itemize}

\noindent\textbf{Edges.} $E = E_{hasGeom} \cup E_{fk} \cup E_{topo} \cup
E_{metric} \cup E_{knn} \cup E_{route} \cup E_{safety} \cup E_{unit}$.
$E_{route}$ connects intent vertices to predicate-class subsets (e.g.\
\texttt{spatial\_measurement} routes to metric predicates).
$E_{unit}$ encodes the geography-cast rule: for any metric predicate on a
geo\_entity with SRID in a lat/lon range, apply \texttt{::geography} to yield
metres instead of degrees.

\noindent\textbf{Materialisation.} $G$ is materialised at agent startup from
three sources: (i)~\texttt{geometry\_columns} for $E_{geo}$ vertices and
\texttt{has\_geometry} edges; (ii)~\texttt{information\_schema.key\_column\_usage}
for foreign-key edges; (iii)~a static registry of OGC SFA predicate classes
and intent routes (algorithm~\ref{alg:build-graph}). This makes $G$
automatically re-materialisable against any new PostGIS schema, supporting
cold-start ingestion in the production data-agent.

\noindent\textbf{Executable subset, not a complete ontology.} $G$ is not a
complete geospatial ontology and is not a GeoSPARQL~\cite{ogcgeosparql}
KB. It is an \emph{executable subset} of the OGC SFA predicate taxonomy
expressed as prompt-side evidence for an LLM-driven NL2GeoSQL agent. Full
ontology alignment (OGC SFA, ISO 19107, GeoSPARQL) is left to future work;
Section~\ref{sec:alignment} documents the rule-level mapping we already
enforce, demonstrating that each semantic-layer rule corresponds to a
specific OGC SFA clause rather than borrowing the standard only by name.
```

```latex
\paragraph{Execution accuracy evaluator.}
EX compares the result set of predicted SQL against gold SQL under the
following rules: (i)~\emph{row order} is ignored via set comparison of
normalised tuples; (ii)~\emph{column order} follows the SELECT list
positional order and is significant; (iii)~\emph{numeric tolerance}:
absolute difference $|a-b| \le 10^{-3}$ OR relative $|a-b|/\max(|a|,|b|) \le
10^{-3}$; (iv)~\emph{NULL comparison}: NULL equals NULL, and NULL differs
from any value; (v)~\emph{floats} are rounded to 3 decimal places prior to
set comparison; (vi)~\emph{LIMIT} results are treated as-is --- if gold
specifies LIMIT N, predicted must also produce $\le N$ rows in the same
sort order. The evaluator is the single function \texttt{compare\_results()}
in \texttt{run\_cq\_eval.py} and is released unchanged in the companion
repository.
```
